4   import jmtetra.tetralogic.Type;
5   import static jmtetra.tetralogic.Type.Squareshape;

7   public class SquareShape extends Tetronome {
8   
9       private Type type;
10      public Piece[] pieces;
11  
12      public SquareShape(Piece[] pieces) {
13          super(pieces);
14          this.pieces = pieces;
15          this.type = Squareshape;
16          createTetronome();
17      }
18  
19      @Override
20      public final void createTetronome() {
21          int yCord = 0;
22          int xCord = 4;
23          for (int i = 0; i < 4; i++) {
24              Piece p = new Piece(yCord, xCord, '0');
25              this.pieces[i] = p;
26  
27              xCord++;
28              if (i == 1) {
29                  yCord++;
30                  xCord = 4;
31              }
32          }
33  
34      }
