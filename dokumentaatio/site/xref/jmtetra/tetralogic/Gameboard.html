2   package jmtetra.tetralogic;

4   import jmtetra.tetralogic.tetronomes.Tetronome;

6   /**
7    * Tämä luokka vastaa pelilaudan luomisesta, ylläpidosta ja päivittämisestä
8    * tarvittaessa.
9    *
10   * @author mikko
11   */
12  public class Gameboard {
13  
14      private int height;
15      private int width;
16      private char[][] gameboard;
17      private Tetronome tetroInPlay;
18  
19      /**
20       * Konstruktorissa luodaan uusi Gameboard, suomeksi pelilauta. Pelilauta on
21       * char[][], eli kaksiulotteinen taulukko, jossa on char muuttujia.
22       */
23      public Gameboard() {
24          this.width = 10;
25          this.height = 16;
26          this.tetroInPlay = null;
27  
28          this.gameboard = new char[height][width];
29          for (int i = 0; i < height; i++) {
30              for (int j = 0; j < width; j++) {
31                  gameboard[i][j] = '-';
32              }
33          }
34      }
35  
36      /**
37       * Metodi saa parametrinaan Tetronomin, joka lisätään pelilautaan.
38       *
39       * @see Tetronome.getPieces()
40       * @param t saa parametrina pelilautaan lisättävän Tetronome-luokan
41       * ilmentymän
42       */
43      public void addTetronome(Tetronome t) {
44          this.tetroInPlay = t;
45  
46          for (Piece p : t.getPieces()) {
47              gameboard[p.getY()][p.getX()] = p.getMark();
48          }
49      }
50  
51      /**
52       * Metodi välittää tiedon (boolean muuttujan) siitä, onko pala osunut
53       * johonkin toiseen palaan. Tämän perusteella pelilooppi osaa tehdä oikean
54       * valinnan seuraavasta toiminnosta.
55       *
56       * @return boolean = onko kierros päättynyt
57       */
58      public boolean isRoundOver() {
59          Piece a = tetroInPlay.getLowestLocation();
60          if (a.getY() == 15) {
61              return true;
62          }
63  
64          Piece[] taulukko = this.getCurTetro().getPieces();
65          for (Piece p : taulukko) {
66  
67              if (!samePieceInBothArrays(new Piece(p.getY() + 1, p.getX(), '-'))) {
68                  if (getChar(p.getY() + 1, p.getX()) != '-') {
69                      return true;
70                  }
71              }
72  
73          }
74  
75          return false;
76  
77      }
78  
79      /**
80       * Tämä metodi tarkistaa onko koko peli loppunut. Käytännössä katsotaan,
81       * onko pala ylimmällä rivillä, jos se osuu jo toiseen palaan (isRoundOver =
82       * true)
83       *
84       * @return boolean = onko peli loppunut
85       */
86      public boolean isGameOver() {
87          if (isRoundOver()) {
88              for (Piece p : tetroInPlay.getPieces()) {
89                  if (p.getY() == 1) {
90                      return true;
91                  }
92              }
93  
94          }
95  
96          return false;
97      }
98  
99      public Tetronome getCurTetro() {
100         return this.tetroInPlay;
101     }
102 
103     public int getHeight() {
104         return height;
105     }
106 
107     public int getWidth() {
108         return width;
109     }
110 
111     public char[][] getGameboard() {
112         return gameboard;

114 /**
115  * Palauttaa halutun character-muuttujan tietystä koordinaatista taulukossa.
116  * @param y y-koordinaatin arvo
117  * @param x x-koordinaatin arvo
118  * @return  character tietystä koordinaatista
119  */
120     public char getChar(int y, int x) {
121         return this.gameboard[y][x];
122     }
123     
124     /**
125      * Asettaa tiettyyn taulukon kohtaan halutun character-muuttujan.
126      * @param y y-koordinaatin arvo
127      * @param x x-koordinaatin arvo
128      * @param c tiettyyn koordinaattiin asetettava character-muuttuja
129      */
130 
131     public void setChar(int y, int x, char c) {
132         this.gameboard[y][x] = c;
133     }
134 
135     /**
136      * Metodi tarkistaa kaikki täydet rivit laudasta.
137      *
138      * @return lukumäärän täysistä riveistä
139      */
140     public int checkAllRowsForFullOnes() {
141         int rowsDestroyed = 0;
142         for (int i = 0; i < height; i++) {
143             if (checkIfAFullRow(i)) {
144                 moveAllPiecesDown(i);
145                 rowsDestroyed++;
146             }
147         }
148         return rowsDestroyed;
149     }
150 
151     /**
152      * Alimetodi, jota ylläoleva metodi checkAllRowsForFullOnes kutsuu.
153      *
154      * @param height eli tarkistettavan rivin sijainti
155      * @return boolean = onko rivi täynnä
156      */
157     public boolean checkIfAFullRow(int height) {
158         for (int i = 0; i < width; i++) {
159             if (getChar(height, i) == '-') {
160                 return false;
161             }
162         }
163         return true;
164     }
165 
166     /**
167      * Metodi tarkastaa onko uuden sijainnin koordinaateissa jo olemassa toista
168      * tetronomea.
169      *
170      * @param newLoc koordinaatit, johon nykyistä pelissä olevaa tetronomea
171      * halutaan siirtää
172      * @return boolean = onko missään uuden sijainnin palassa jo toista palaa
173      */
174     public boolean checkIfOccupied(Piece[] newLoc) {
175         for (Piece p : newLoc) {
176             if (samePieceInBothArrays(p)) {
177                 continue;
178             }
179             if (p.getX() >= this.getWidth() || p.getX() < 0) {
180                 return false;
181             }
182             if (p.getY() >= this.getHeight() || p.getY() < 0) {
183                 return false;
184             }
185             if (this.getChar(p.getY(), p.getX()) != '-') {
186                 return false;
187             }
188         }
189         return true;
190 
191     }
192 
193     /**
194      * Tämä metodi tarkastaa onko alempana/sivulla oleva pala osa samaa
195      * tetronomea kuin pelissä oleva pala Käytännössä mahdollistetaan ettei tule
196      * vääriä ilmoituksia sijainnissa olevasta palasta, joka onkin vain osa
197      * tetronomea, jota halutaan liikuttaa
198      *
199      * @param p pala, joka tarkastetaan
200      * @return boolean = onko pala osa nykyistä tetronomea
201      */
202     private boolean samePieceInBothArrays(Piece p) {
203         for (Piece pi : this.tetroInPlay.getPieces()) {
204             if (p.getX() == pi.getX() && p.getY() == pi.getY()) {
205                 return true;
206             }
207         }
208         return false;
209     }

211     /**
212      * Ehkäpä luokan tärkein metodi, joka päivittää aina lautaa vastaamaan
213      * nykyistä pelitilannetta.
214      *
215      * @param newLoc uuden palan sijainnin omaava Piece[] taulukko
216      */
217 
218     public void updateBoard(Piece[] newLoc) {
219         if (this.checkIfOccupied(newLoc)) {
220             Piece[] oldCoords = tetroInPlay.getPieces();
221 
222             for (Piece p : oldCoords) {
223                 setChar(p.getY(), p.getX(), '-');
224             }
225 
226             for (Piece p : newLoc) {
227                 setChar(p.getY(), p.getX(), p.getMark());
228             }
229 
230             tetroInPlay.setPieces(newLoc);
231 
232         }
233     }

235     /**
236      * Liikuttaa kaikkia laudassa olevia paloja yhden askeleen alaspäin Tämä
237      * metodi toteutetaan, jos löytyy täynnä olevia rivejä, joita tuhotaan.
238      *
239      * @param startingHeight aloituskorkeus, joka/josta ylöspäin kaikkia rivejä
240      * siirretään alaspäin
241      */
242     public void moveAllPiecesDown(int startingHeight) {
243 
244         for (int y = startingHeight; y > 0; y--) {
245             for (int x = 0; x < width; x++) {
246                 setChar(y, x, getChar(y - 1, x));
247             }
248         }
249     }
